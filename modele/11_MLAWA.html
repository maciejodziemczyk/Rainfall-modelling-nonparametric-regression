def normality_test(model, alpha = 0.05, rescaled_moments = False):
    '''
    Test jarque-bera
    model - oszacowany model, dla ktrego ma by policzona statystyka
    alpha - poziom istotnoci, dla ktrego liczony jest przedzia ufnoci
    rescaled_moments - bools/ jeeli True obliczana jest statystyka dostosowana do wynikw regresji (przeskalowane momenty)
        A.H.M.R Imon, Intorduction to Regression residuals, moments, and their use in tests for normality,
        Communications in Statistics - Theory and Methods, 32 (2003), 1021 - 1034
        
    return (statystyka tesotwa, warto krytyczna, p_value)
    '''
       
    # import bibliotek
    from scipy.stats import kurtosis
    from scipy.stats import skew
    from scipy.stats import chi2
    
    # pozyskiwanie zmiennych
    n_obs = len(model.resid)                       # liczba obserwacji
    S = skew(model.resid)                          # skono
    K = kurtosis(model.resid, fisher = True)       # kurtoza (uwzgldnia -3 w definicji)
    
    
    if rescaled_moments:
        c = n_obs/model.df_resid                   # parametr przeskalowania momentw 
        RM = (n_obs*(c**3)/6)*(S**2 + c*(K**2)/4)    # statystyka testowa RM
    
        critical_value = chi2.ppf(q = 1-alpha, df = 2) # warto krytyczna
        p = 1-chi2.cdf(RM, df = 2)                     # p value

        return (RM, critical_value, p)
    
    else:
        JB = (n_obs/6)*(S**2 + (K**2)/4)           # statystyka testowa oglna
    
        critical_value = chi2.ppf(q = 1-alpha, df = 2) # warto krytyczna
        p = 1-chi2.cdf(JB, df = 2)                     # p value

        return (JB, critical_value, p)

def get_data(station, target, features, data, control_zero=None):
    '''
    Wydobycie z bazy danych obserwacji dotyczycych wybranej stacji, argumenty:
    station - stacja, dla ktrej maj zwrcone zosta dane,
    target - zmienna zalena (Y),
    features - zmienne niezalene (X),
    data - baza danych,
    control_zero - zmienna po ktrej naley kontrolowa zera (opcjonalne)
    
    return X, Y 
    '''
    
    # --- zbudowanie zbioru danych do analizy --- 
    # znalezienie kodu dla stacji o podanej nazwie (nazwy zmieniay si na przestrzeni lat dlatego lepiej posugiwa si kodem)
    code = data["station_code"].where(data["station_name"]==station).dropna()
    if code.value_counts().shape[0] > 1:                                       # bd jeeli wicej ni 1 kod dla stacji
        raise "Znaleziono rne kody dla podanej stacji"
    code = code.to_numpy().max()                                               # wybranie jednej wartoci (mogaby by inna funkcja ni max)
    
    # macierze Y i X (jako dataframe) dla konkretnej stacji i zmiennych "endogenicznych" i egzogenicznych
    Y=pd.DataFrame(data[target].loc[data["station_code"]==code])
    X=data[features].loc[data["station_code"]==code]
    
    if control_zero and (control_zero not in features):
        raise "control_zero musi zawiera si w features"
    
    if control_zero and (control_zero in features):
        indeksy = X.where(X[control_zero]<=0).dropna().index.tolist()
        
        if len(indeksy) >= 2:                                                  # jeeli jest len(indeksy) < 2 to dane ok
            for (i, i1) in zip(indeksy, indeksy[1:]):
                if i + 1 != i1:
                    #print("i: {}, i1: {}".format(i,i1))
                    break
            X = X.loc[i:,:]
            Y = Y.loc[i:,:]
        
    return X, Y

def train_test_split(X, Y, test_percentage):
    '''
    funkcja dzielca zbir danych na zbir testowy i treningowy, argumenty:
    X - macierz zmiennych objaniajcych (Dataframe),
    Y - wektor zmiennych objanianych (Dataframe),
    test_percentage - ile procent zmiennych ma si znale w zbiorze testowym 
    
    return X_test, Y_test, X_train, Y_train
    
    ---------------------------------------
    Note: w prbie testowej znajduje si k ostatnich obserwacji ze zbioru (po indeksie)
    '''
        
    # liczba obserwacji w caym zbiorze
    n = X.shape[0]
    # liczba obserwacji w zbiorze testowym
    n_test = round(n*test_percentage)
    # pozyskanie indeksw wszystkich obserwacji
    X_index = X.index.tolist()
    # rozdzielenie indeksw na indeksy testowe i treningowe
    X_test_index = X_index[-n_test:]                          
    X_train_index = [i for i in X_index if i not in X_test_index]
    
    # zbudowanie testowych i treningowych zbiorw 
    X_test = X[X.index.isin(X_test_index)]
    X_train = X[X.index.isin(X_train_index)]
    Y_test = Y[Y.index.isin(X_test_index)]
    Y_train = Y[Y.index.isin(X_train_index)]
    
    
    return X_test, Y_test, X_train, Y_train

def get_metrices(model, Y, lmbda = None):
    '''
    obliczenie podstawowych metryk dla modelu, argumenty:
    model - model, dla ktrego liczymy metryki,
    y - wartoci zaobserwowane,
    lmbda - warto lambda (opcjonalne - jeeli model liczony by na transformacji potgowej Y),
    X_test - zbir obserwacji testowych, jeeli predict == True,
    predict - True/False - czy metryki maj by liczone dla predykcji
    
    return metrices (DataFrame)
    '''
    from scipy.special import inv_boxcox
   
    if lmbda:
        y_hat = inv_boxcox(model.fittedvalues.values, lmbda)
    else:
        y_hat = model.fittedvalues.values
        
    y = Y.values.ravel()
    n = y.shape[0]

    mse = 1/n*np.square(y-y_hat).sum()
    rmse = mse**(1/2)
    mae = 1/n*np.abs(y-y_hat).sum()
    mape = 1/n*np.sum(np.abs((y-y_hat)/y))*100
    
    metrices = pd.DataFrame(data = [mse, rmse, mae, mape], index = ["mse", "rmse", "mae", "mape"]).T
    
    return metrices

def OLS(X, Y, lag=0, ar = False, box_cox = False, get_vif = False, get_corr = False):
    '''
    funkcja zwracajca model pogodowy, argumenty:
    station - stacja dla ktrej obliczany jest model (OLS)
    target - zmienna objaniana
    features - zmienne objaniajce
    lag - opnienie zmiennych objaniajcych (domylnie 0)
    ar - True/False - dodanie komponentu autoregresyjnego AR(p), p = lag
    box_cox - True/False - optymalna transformacja boxa coxa
    get_vif - True/False, zwraca dataframe z VIFami (funkcja zwraca o 1 obiekt wicej)
    get_corr - True/False zwraca macierz korelacji spearmana jako dataframe
    data - zbir danych, na ktrym ma by zastosowana funkcja
    
    return (optional(lambda), model, optional(VIF), optional(spearman correlation matrix))
    '''
    
    # biblioteki do OLS, VIF, spearmana, boxa-coxa
    import statsmodels.formula.api as smf
    from statsmodels.stats.outliers_influence import variance_inflation_factor
    from scipy.stats import spearmanr, boxcox
    import warnings
    
    
    # lista zawierajca obiekty do zwrotu
    zwroty = [] 
    target = Y.columns.tolist()[0]

    # --- transformacja boxa-coxa ---    
    if box_cox:
        y_min = Y.min()[0]
        if y_min <= 0:                                          
            Y = Y + (abs(y_min) + 10**(-10))   # usunicie ujemnych wartoci
            #zwroty.append(y_min)
            warnings.warn("dane zostay powikszone -> \nY = Y + (abs(y_min) + 10**(-10)) ")
            #print("dane zostay powikszone o {}".format(y_min))
        Y_box_cox, l = boxcox(Y.values.ravel())
        Y = pd.Series(Y_box_cox, index = Y.index, name = target)
        zwroty.append(l)
        #zwroty.append(Y)
    
    # --- estymacja modelu w zalenoci od wariantu opnie ---
    if lag > 0:
        
        if ar:
            # zczenie macierzy Y i X
            X = pd.concat((Y,X), axis=1)
            # zmiana nazwy zmiennej autoregresyjnej
            X.rename(mapper = {X.columns.tolist()[0]:"{}_lag_{}".format(target, lag)},
                     axis = 1, inplace = True)
            
        # opnienie macierzy X o lag okresw
        X = X.shift(periods = lag)
        # usunicie najstarszej obserwacji dla ktrej nie ma wartoci opnionej -> pojawia si NaN
        for i in range(lag):
            X = X.drop(labels = np.min(X.index))
        # usunicie najstarszej obserwacji dla Y (koresponduje z X)
        for i in range(lag):
            Y = Y.drop(labels = np.min(Y.index))
        
        # oszacowanie modelu z opnieniami
        dane = pd.concat((Y,X), axis = 1)
        formula = "{}~".format(target) + '+'.join(X.columns.tolist())
        model = smf.ols(formula = formula, data = dane).fit()
        
    
    else:
        # oszacowanie modelu bez opnie
        dane = pd.concat((Y,X), axis = 1)
        formula = "{}~".format(target) + '+'.join(X.columns.tolist())
        model = smf.ols(formula = formula, data = dane).fit()
    
    # aktualizacja listy do zwrotu
    zwroty.append(model)
    
    # --- obliczenie VIFw i zwrot wynikw ---
    if get_vif:
        # obliczenie VIFw
        vif = np.round([variance_inflation_factor(X.values, i) for i in range(X.shape[1])], 2)

        # przystosowanie wynikw do wywietlenia
        vif = list(map(float, vif))
        name = list(X)
        s1 = pd.Series(name, name = 'Zmienna')
        s2 = pd.Series(vif, name = 'VIF')
        VIF_results = pd.concat([s1,s2], axis = 1)
        
        # aktualizacja listy do zwrotu
        zwroty.append(VIF_results)
    
    # --- obliczenie macierzy korelacji spearmana ---
    if get_corr:
        correl = np.round(data[X.columns.tolist()].corr(method = 'spearman'), 2)
        # aktualizacja listy do zwrotu
        zwroty.append(correl)


    return tuple(zwroty)
